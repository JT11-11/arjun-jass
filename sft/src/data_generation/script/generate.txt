## OpenAI API Key

export OPENAI_API_KEY=""

## To generate the games

python3 src/main.py \
  --total_examples 264 \
  --altruism_ratio 0.95 \
  --output_file data/production_training.jsonl \
  --seed 33

## To Upload to OpenAI 
curl https://api.openai.com/v1/files \
  -H "Authorization: Bearer $OPENAI_API_KEY" \
  -F purpose="fine-tune" \
  -F file="@data/production_training.jsonl"

## Validation Sets (Optional)
curl https://api.openai.com/v1/files \
  -H "Authorization: Bearer $OPENAI_API_KEY" \
  -F "purpose=fine-tune" \
  -F "file=@validation.jsonl"

## To Start SFT Training
curl https://api.openai.com/v1/fine_tuning/jobs \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $OPENAI_API_KEY" \
  -d '{
    "training_file": "file-5z7qcRYpNLPxSMQ3ujd5ut",
    "model": "gpt-3.5-turbo-0125"
  }'

## Check SFT Status
